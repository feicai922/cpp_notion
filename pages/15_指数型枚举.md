## 16.指数型枚举
```
#include<bits/stdc++.h>
using namespace std;
int n;
vector<vector<int>> ans;
vector<int>temp;
void dfs(int x)
{
    if(x>n)
    {
        ans.push_back(temp);
        return;
    }
    // 不选 x
    dfs(x+1);
    // 选 x
    temp.push_back(x);
    dfs(x+1);
    // 回溯
    temp.pop_back();
}
int main()
{
    cout<<" ";
    cin>>n;
    // 从第 1 个元素开始枚举
    dfs(1);
    for(auto &subset:ans)
    {
        for (int num: subset)
            cout << num << " ";
        cout<<endl;
    }
    return 0;
}
```
