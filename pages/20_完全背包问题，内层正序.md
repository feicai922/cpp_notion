## 21.完全背包问题，内层正序
```
// 和 01背包的区别就在于内层循环的正序的
#include<bits/stdc++.h>
using namespace std;
int n,m;
typedef struct node
{
    int value,weight;
}bag;
int main()
{
    cin>>n>>m;
    bag a[n+1];
    for(int i=1;i<=n;i++)
        cin>>a[i].weight>>a[i].value;
    int dp[m+1];
    memset(dp,0,sizeof dp);
    for(int i=1;i<=n;i++)
    {
        // 正序遍历，可以使得每个物品被重复选择
        for(int j=a[i].weight;j<=m;j++)
            dp[j] = max(dp[j],dp[j-a[i].weight]+a[i].value);
    }
    cout<<dp[m]<<endl;
    return 0;
}
```
