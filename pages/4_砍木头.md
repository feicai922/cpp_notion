## 5.砍木头
https://www.luogu.com.cn/problem/P2440
```
#include<bits/stdc++.h>
using namespace std;
long long n,k;
long long a[1000005];
bool f(long long x)
{
    long long ans = 0;
    for(int i=0;i<n;i++)
    {
        ans += a[i]/x;
    }
    return ans >= k;
}
int main()
{
    cin>>n>>k;
    for(int i=0;i<n;i++) scanf("%d",&a[i]);
    long long l = 0,r = 100000001, mid;、
    // while(l + 1 < r) 的条件确保了在循环结束时，l 和 r 相邻（即 r = l + 1）
    while(l+1<r)
    {
        mid = (l+r)/2;
        if(f(mid)) l = mid;
        else r = mid;
    }
    cout<<l<<endl;
    return 0;
}
```
