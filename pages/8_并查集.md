## 9.并查集
```
#include <bits/stdc++.h>
using namespace std;
const int N = 1e5 + 10;
int pre[N], h[N]; // h数组用于记录每个连通块的大小

void init(int n)
{
    for (int i = 1; i <= n; i++)
    {
        pre[i] = i; // 初始化每个节点的父节点为自身
        h[i] = 1; // 每个节点初始时是一个独立的连通块，大小为1
    }
}

int find(int x)
{
    if (pre[x] == x)
        return x;
    return pre[x] = find(pre[x]); // 路径压缩
    // attention！一定是等号，而不是==
    // 上一句也可以是return find(pre[x])
}

bool same(int a, int b)
{
    return find(a) == find(b); // 判断两个节点是否在同一连通块
}

void join(int a, int b)
{
    a = find(a);
    b = find(b);
    if (a == b)
        return; // 如果已经在同一个连通块，直接返回
    if (h[a] > h[b]) // 合并时，将较小的连通块合并到较大的连通块
    {
        pre[b] = a;
        h[a] += h[b]; // 更新连通块大小
    }
    else
    {
        pre[a] = b;
        h[b] += h[a];
    }
}

int main()
{
    int n, m;
    cin >> n >> m;
    init(n);
    while (m--)
    {
        string op;
        int a, b;
        cin >> op;
        if (op == "C")
        {
            cin >> a >> b;
            join(a, b);
        }
        else if (op == "Q1")
        {
            cin >> a >> b;
            if (same(a, b))
                cout << "Yes" << endl;
            else
                cout << "No" << endl;
        }
        else
        {
            cin >> a;
            cout << h[find(a)] << endl; // 直接输出连通块的大小
        }
    }
    return 0;
}
```
