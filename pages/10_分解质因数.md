## 11.分解质因数
```
#include<bits/stdc++.h>
using namespace std;
const int N = 3e5+10;
map<int,bool>mp;
void init() // 初始化素数表
{
    for(int i=1;i<=N;i++)
        mp[i]=true;
    mp[1]=false;
    for(int i=2;i*i<=N;i++)
    {
        if(mp[i])//i是素数，i的倍数不是素数
        {
            for(int j=i*i;j<=N;j+=i)
                mp[j]=false;
        }
    }
}
void process(int x)
{
    if(mp[x])
    {
        cout << x << "=" << x << endl;
        return;
    }
    else
        cout<<x<<"=";
    vector<int>a;
    for(int i=2;i*i<=x;i++)
    {

        while(x%i==0&&mp[i])
        {
            a.push_back(i);
            x/=i;
        }
    }
    // 确保在分解过程中，如果x被除到小于i * i时，剩余的x（如果大于1）被正确处理
    if(x>1) a.push_back(x);
    for(int i=0;i<a.size();i++)
    {
        cout << a[i];
        if (i < a.size() - 1) cout << "*";
    }
    cout<<endl;
}
int main()
{
    init();
    int a,b;
    cin>>a>>b;
    for(int i=a;i<=b;i++)
        process(i);
    return 0;
}
```
